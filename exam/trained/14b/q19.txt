```ts
import { test, expect } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';

test('WCAG 2.2 AA only audit', async ({ page }) => {
  await page.goto('https://agency.example.org/accessibility-test');

  const results = await new AxeBuilder({ page })
    .withTags(['wcag2a', 'wcag2aa'])
    .disableRules(['color-contrast', 'duplicate-id']) // optional: suppress known false positives
    .analyze();

  expect(results.violations).toHaveLength(0);
});
```
