```python
import fitz  # PyMuPDF
from pathlib import Path

def has_text_layer(pdf_path: str) -> list[str]:
    """Return list of lines indicating whether each page has a text layer."""
    results: list[str] = []
    doc = fitz.open(pdf_path)
    for i, page in enumerate(doc):
        text = page.get_text("text")
        if text.strip():
            results.append(f"page {i + 1}: text")
        else:
            # Check for text in blocks (more robust than get_text(""))
            blocks = page.get_text("blocks")
            if any(b[4].strip() for b in blocks):
                results.append(f"page {i + 1}: text")
            else:
                results.append(f"page {i + 1}: image-only")
    doc.close()
    return results

# Example usage
if __name__ == "__main__":
    path = Path("example.pdf")
    if path.exists():
        for line in has_text_layer(str(path)):
            print(line)
    else:
        print(f"File not found: {path}")
```
