```ts
import { test, expect } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';

test('dashboard is accessible after login', async ({ page }) => {
  // 1. Navigate to login page and fill form
  await page.goto('https://agency.example.gov/login');
  await page.getByLabel('Username').fill('test@example.gov');
  await page.getByLabel('Password').fill('placeholder');
  await page.getByRole('button', { name: 'Log in' }).click();

  // 2. Wait for dashboard to load
  await page.waitForURL('**/dashboard');
  await expect(page.getByRole('heading', { name: 'Dashboard' })).toBeVisible();

  // 3. Run axe audit
  const results = await new AxeBuilder({ page })
    .withTags(['wcag2a', 'wcag2aa', 'wcag22aa'])
    .include('main')
    .disableRules(['color-contrast', 'duplicate-id']) // Example: suppress known false positives
    .analyze();

  // 4. Assert no violations
  expect(results.violations).toEqual([]);
});
```
